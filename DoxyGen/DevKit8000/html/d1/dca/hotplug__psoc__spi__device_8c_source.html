40  /* Add the slave SPI device to the SPI bus
41  *
42  * These methods are used to hot-plug spi devices.
43  * SPI devices are by nature NOT hot-pluggable, as
44  * they cannot be probed for functionality etc. SPI
45  * devices are normally cold-plugged during boot, that
46  * is, they are added in the board description file:
47  * /arch/arm/mach-omap2/board-devkit8000.c
48  * Using this method we actually doing "hot" cold-plugging
49  * adding devices using a kernel module.
50  * Note that it is crusial that driver and device uses
51  * the same name alias. If not, the device and driver
52  * will not be paired and the probe method in the driver
53  * not be called.
54  */
55  bus_num = slave_spi_board_info.bus_num;
56  slaves_spi_master = spi_busnum_to_master(bus_num);
57  slave_spi_device = spi_new_device(slaves_spi_master,
58  &slave_spi_board_info);
59  if(slave_spi_device < 0) {
60  printk(KERN_ALERT "Unsuccesful creating a new device\n");
61  return -1;
62  }
63 
64  return 0;
65 }
